```cs
//  Types:
char
string
int
float
double
decimal
bool

// Variables
char symbol = 'x';
string text = "Hello world!";
int number = 5;
float number2 = 4.58;

bool isOpen = false;

// Arithmetic Operators
+  // sum
-  // difference
*  // multiplication
/  // division
%  // remainder (modulus)
++ // increment (increase by 1)
-- // decrement (decrease by 1)

double num = 10;

num + 3  // 13
num - 4  // 6
num * 7  // 70
num / 4  // 2.25
num % 3  // 1
num++    // 11
num--    // 9

// Comparison Operator
<  // less than
>  // greater than
<= // less than or equal to
>= // greater than or equal to
== // equal to
!= // not equal to

1 < 6     // true
5 > 7     // false
5 <= 5    // true
7 >= 9    // false
99 == 66  // false
73 != 7   // true
```
